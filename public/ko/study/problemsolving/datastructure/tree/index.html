#include <vector>
#include <queue>
using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);

	int n, node1, node2;//ì •ì ì˜ ê°œìˆ˜
	int p[1000001] = {};
	vector<int> v[1000001];
	queue<int> q;
	cin >> n;
	for (int i = 0; i < n - 1; i++)
	{
		cin >> node1 >> node2;
		v[node1].push_back(node2);
		v[node2].push_back(node1);
	}
	q.push(1);
	while (!q.empty())
	{
		int cur = q.front();
		q.pop();

		for (auto nxt : v[cur])
		{
			if (p[cur] == nxt)	continue;
			q.push(nxt);
			p[nxt] = cur;
		}
	}
	for (int i = 2; i <= n; i++)
		cout << p[i] << '\n';

	return 0;
}
